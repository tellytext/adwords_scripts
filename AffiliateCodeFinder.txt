function main() {
  MccApp.accounts().executeInParallel("checkURLs", "");
}

function checkURLs() {
  
  var STR_TO_FIND = "365_690669"
  var keywords = AdWordsApp.keywords().get();
  
  Logger.log("Checking keywords for account %s", AdWordsApp.currentAccount().getName());
  
  while (keywords.hasNext()) {
    var keyword = keywords.next();
    var keywordMobileUrl = keyword.urls().getMobileFinalUrl();
    var keywordDesktopUrl = keyword.urls().getFinalUrl();
    var keywordTrackingTemplate = keyword.urls().getTrackingTemplate();
    
    if (keywordMobileUrl !== null) {
      if(keywordMobileUrl.indexOf(STR_TO_FIND) > -1) {
        Logger.log("%s %s %s", keyword.getText(), keyword.getId(), keywordMobileUrl)
      }
    }
    if (keywordDesktopUrl !== null) {
      if(keywordDesktopUrl.indexOf(STR_TO_FIND) > -1) {
        Logger.log("%s %s %s", keyword.getText(), keyword.getId(), keywordDesktopUrl)
      }
    }
    if (keywordTrackingTemplate !== null) {
      if(keywordTrackingTemplate.indexOf(STR_TO_FIND) > -1) {
        Logger.log("%s %s %s", keyword.getText(), keyword.getId(), keywordTrackingTemplate)
      }
    }
  }
}
